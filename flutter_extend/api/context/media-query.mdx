### üìê Screen Dimensions
Utilities to get screen size without strict `MediaQuery` boilerplate.

{/*  Screen Size */}
<ResponseField name="screenSize" type="double">
The total size of the screen. Equivalent to `MediaQuery.sizeOf(context).`

<CodeBlock
    filename="context"
    language="dart"
    icon="water"
    wrap="true"
    lines="true"
    expandable="false"
>
    final width = context.screenSize;
</CodeBlock>
</ResponseField>

{/*  Screen Width */}
<ResponseField name="screenWidth" type="double">
The width of the screen in logical pixels.

<CodeBlock
    filename="context"
    language="dart"
    icon="water"
    wrap="true"
    lines="true"
    expandable="false"
>
    final width = context.screenWidth;
</CodeBlock>

</ResponseField>

{/*  Screen Height */}
<ResponseField name="screenHeight" type="double">
The height of the screen in logical pixels.

<CodeBlock
    filename="context"
    language="dart"
    icon="water"
    wrap="true"
    lines="true"
    expandable="false"
>
    // Dynamically size a container to half the screen height
    Container(height: context.screenHeight * 0.5);
</CodeBlock>
</ResponseField>

{/*  Dark Mode */}
<ResponseField name="isDarkMode" type="bool">

Returns `true` if the system theme is dark.

<CodeBlock
    filename="context"
    language="dart"
    icon="water"
    wrap="true"
    lines="true"
    expandable="false"

</CodeBlock>
</ResponseField>

    icon="water"
    wrap="true"
    lines="true"
    expandable="false"

{/*  Keyboard Visibility */}

    icon="water"
    wrap="true"
